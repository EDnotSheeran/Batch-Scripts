echo off/on            = desliga ou liga as msm gos sistema 
vol C:                 = da informacoes sobre o volume expecificado
ver                    = diz que versao do sistema operacional vc esta usando
clip                   = copia o restutado de um comando para a area de transferencia
echo hello | clip      = vai fazer com que hello seja copiado para a area de transferencia
date                   = pergunta a data para altera a hora do sistema se digitado sozinho
date 01/01/2017        = altera a data para 01/01/2017
date /t                = te informa a data do sistema
time                   = igual o date mais com as horas e minutos
start                  = inicia outras tarefas
start notepad          * inicia o notepad
start www.google.com   * abre o site com o navegador padrao
start mrt              * abre a ferramanta de romocao  de sodtware mal intencionado da microsoft
start "titulo" cmd.exe * abre outra instancia do cmd com o titulo declarado
winver                 = versao grafica do ver
